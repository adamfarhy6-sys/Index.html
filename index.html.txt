  <script>
    const canvas = document.getElementById('gameCanvas');
    const ctx = canvas.getContext('2d');

    // Responsive sizing
    const maxSize = Math.min(window.innerWidth - 48, 360);
    const CELL = 18;
    const COLS = Math.floor(maxSize / CELL);
    const ROWS = Math.floor(maxSize / CELL);
    canvas.width = COLS * CELL;
    canvas.height = ROWS * CELL;

    // Game state
    let snake, dir, nextDir, food, specialFood, score, best, level, gameLoop, paused, gameOver;
    let specialFoodTimer = 0;
    let specialFoodActive = false;
    let particles = [];

    // Level config: [baseInterval, pointsPerFood, levelUpEvery]
    const LEVELS = [
      { interval: 160, color: '#00ff88' },
      { interval: 135, color: '#00e077' },
      { interval: 115, color: '#00ccff' },
      { interval: 100, color: '#ff9900' },
      { interval: 88,  color: '#ff4466' },
      { interval: 78,  color: '#cc44ff' },
      { interval: 70,  color: '#ff44cc' },
      { interval: 62,  color: '#ff2200' },
      { interval: 56,  color: '#ffff00' },
      { interval: 50,  color: '#ffffff' },
    ];
    const LEVEL_UP_SCORE = 50;

    // Load best score
    best = parseInt(localStorage.getItem('snake_best') || '0');
    document.getElementById('best').textContent = best;

    function getLevelData() {
      return LEVELS[Math.min(level - 1, LEVELS.length - 1)];
    }

    function initGame() {
      const midX = Math.floor(COLS / 2);
      const midY = Math.floor(ROWS / 2);
      snake = [
        { x: midX, y: midY },
        { x: midX - 1, y: midY },
        { x: midX - 2, y: midY },
      ];
      dir = { x: 1, y: 0 };
      nextDir = { x: 1, y: 0 };
      score = 0;
      level = 1;
      paused = false;
      gameOver = false;
      particles = [];
      specialFoodActive = false;
      specialFoodTimer = 0;
      placeFood();
      updateUI();
    }

    function placeFood() {
      let pos;
      do {
        pos = {
          x: Math.floor(Math.random() * COLS),
          y: Math.floor(Math.random() * ROWS)
        };
      } while (snake.some(s => s.x === pos.x && s.y === pos.y));
      food = pos;

      // Random chance to spawn special food
      if (Math.random() < 0.25 && !specialFoodActive) {
        placeSpecialFood();
      }
    }

    function placeSpecialFood() {
      let pos;
      do {
        pos = {
          x: Math.floor(Math.random() * COLS),
          y: Math.floor(Math.random() * ROWS)
        };
      } while (
        snake.some(s => s.x === pos.x && s.y === pos.y) ||
        (pos.x === food.x && pos.y === food.y)
      );
      specialFood = pos;
      specialFoodActive = true;
      specialFoodTimer = 80; // ticks before disappearing
    }

    function updateUI() {
      document.getElementById('score').textContent = score;
      document.getElementById('level').textContent = level;
      const progress = (score % LEVEL_UP_SCORE) / LEVEL_UP_SCORE * 100;
      document.getElementById('level-fill').style.width = progress + '%';
      document.getElementById('level-label').textContent = 'LVL ' + level;
      if (score > best) {
        best = score;
        localStorage.setItem('snake_best', best);
        document.getElementById('best').textContent = best;
      }
    }

    function spawnParticles(x, y, color, count = 8) {
      for (let i = 0; i < count; i++) {
        const angle = (Math.PI * 2 / count) * i + Math.random() * 0.5;
        const speed = 1.5 + Math.random() * 2;
        particles.push({
          x: x * CELL + CELL / 2,
          y: y * CELL + CELL / 2,
          vx: Math.cos(angle) * speed,
          vy: Math.sin(angle) * speed,
          life: 1,
          color,
          size: 3 + Math.random() * 3
        });
      }
    }

    function updateParticles() {
      particles = particles.filter(p => p.life > 0);
      for (const p of particles) {
        p.x += p.vx;
        p.y += p.vy;
        p.vy += 0.1;
        p.life -= 0.04;
        p.size *= 0.96;
      }
    }

    function drawParticles() {
      for (const p of particles) {
        ctx.save();
        ctx.globalAlpha = p.life;
        ctx.fillStyle = p.color;
        ctx.beginPath();
        ctx.arc(p.x, p.y, p.size, 0, Math.PI * 2);
        ctx.fill();
        ctx.restore();
      }
    }

    function drawGrid() {
      ctx.strokeStyle = 'rgba(255,255,255,0.03)';
      ctx.lineWidth = 0.5;
      for (let x = 0; x <= COLS; x++) {
        ctx.beginPath();
        ctx.moveTo(x * CELL, 0);
        ctx.lineTo(x * CELL, canvas.height);
        ctx.stroke();
      }
      for (let y = 0; y <= ROWS; y++) {
        ctx.beginPath();
        ctx.moveTo(0, y * CELL);
        ctx.lineTo(canvas.width, y * CELL);
        ctx.stroke();
      }
    }

    function drawFood() {
      const cx = food.x * CELL + CELL / 2;
      const cy = food.y * CELL + CELL / 2;
      const r = CELL / 2 - 2;
      const pulse = Math.sin(Date.now() / 200) * 2;

      // Glow
      ctx.save();
      ctx.shadowColor = '#ff4444';
      ctx.shadowBlur = 10 + pulse;
      ctx.fillStyle = '#ff4444';
      ctx.beginPath();
      ctx.arc(cx, cy, r + pulse * 0.3, 0, Math.PI * 2);
      ctx.fill();
      ctx.restore();

      // Apple highlight
      ctx.fillStyle = 'rgba(255,255,255,0.35)';
      ctx.beginPath();
      ctx.arc(cx - 2, cy - 2, r * 0.35, 0, Math.PI * 2);
      ctx.fill();
    }

    function drawSpecialFood() {
      if (!specialFoodActive) return;
      const cx = specialFood.x * CELL + CELL / 2;
      const cy = specialFood.y * CELL + CELL / 2;
      const r = CELL / 2 - 1;
      const t = Date.now() / 150;

      ctx.save();
      ctx.shadowColor = '#ffdd00';
      ctx.shadowBlur = 14 + Math.sin(t) * 4;

      // Star shape
      ctx.fillStyle = '#ffdd00';
      ctx.beginPath();
      for (let i = 0; i < 5; i++) {
        const angle = (Math.PI * 2 / 5) * i - Math.PI / 2 + Math.sin(t) * 0.2;
        const outerX = cx + Math.cos(angle) * r;
        const outerY = cy + Math.sin(angle) * r;
        const innerAngle = angle + Math.PI / 5;
        const innerX = cx + Math.cos(innerAngle) * r * 0.45;
        const innerY = cy + Math.sin(innerAngle) * r * 0.45;
        if (i === 0) ctx.moveTo(outerX, outerY);
        else ctx.lineTo(outerX, outerY);
        ctx.lineTo(innerX, innerY);
      }
      ctx.closePath();
      ctx.fill();

      // Fade out indicator
      const alpha = specialFoodTimer / 80;
      ctx.globalAlpha = 1 - alpha * 0.5 + 0.3;
      ctx.restore();

      // Timer ring
      ctx.save();
      ctx.strokeStyle = 'rgba(255, 220, 0, 0.5)';
      ctx.lineWidth = 1.5;
      ctx.beginPath();
      ctx.arc(cx, cy, r + 3, -Math.PI / 2, -Math.PI / 2 + Math.PI * 2 * (specialFoodTimer / 80));
      ctx.stroke();
      ctx.restore();
    }

    function drawSnake() {
      const levelColor = getLevelData().color;

      for (let i = 0; i < snake.length; i++) {
        const seg = snake[i];
        const x = seg.x * CELL;
        const y = seg.y * CELL;
        const isHead = i === 0;
        const t = i / snake.length;

        // Gradient from head to tail
        const alpha = 1 - t * 0.5;
        ctx.save();
        ctx.globalAlpha = alpha;

        if (isHead) {
          // Head glow
          ctx.shadowColor = levelColor;
          ctx.shadowBlur = 12;
        }

        // Segment
        const grad = ctx.createLinearGradient(x, y, x + CELL, y + CELL);
        grad.addColorStop(0, levelColor);
        grad.addColorStop(1, shadeColor(levelColor, -40));
        ctx.fillStyle = grad;

        const radius = isHead ? 5 : 3;
        roundRect(ctx, x + 1, y + 1, CELL - 2, CELL - 2, radius);
        ctx.fill();

        // Head eyes
        if (isHead) {
          ctx.shadowBlur = 0;
          ctx.fillStyle = '#000';
          const eyeSize = 2.5;
          if (dir.x === 1) {
            ctx.fillRect(x + CELL - 5, y + 3, eyeSize, eyeSize);
            ctx.fillRect(x + CELL - 5, y + CELL - 5, eyeSize, eyeSize);
          } else if (dir.x === -1) {
            ctx.fillRect(x + 3, y + 3, eyeSize, eyeSize);
            ctx.fillRect(x + 3, y + CELL - 5, eyeSize, eyeSize);
          } else if (dir.y === -1) {
            ctx.fillRect(x + 3, y + 3, eyeSize, eyeSize);
            ctx.fillRect(x + CELL - 5, y + 3, eyeSize, eyeSize);
          } else {
            ctx.fillRect(x + 3, y + CELL - 5, eyeSize, eyeSize);
            ctx.fillRect(x + CELL - 5, y + CELL - 5, eyeSize, eyeSize);
          }
        }

        ctx.restore();
      }
    }

    function roundRect(ctx, x, y, w, h, r) {
      ctx.beginPath();
      ctx.moveTo(x + r, y);
      ctx.lineTo(x + w - r, y);
      ctx.arcTo(x + w, y, x + w, y + r, r);
      ctx.lineTo(x + w, y + h - r);
      ctx.arcTo(x + w, y + h, x + w - r, y + h, r);
      ctx.lineTo(x + r, y + h);
      ctx.arcTo(x, y + h, x, y + h - r, r);
      ctx.lineTo(x, y + r);
      ctx.arcTo(x, y, x + r, y, r);
      ctx.closePath();
    }

    function shadeColor(hex, amount) {
      let r = parseInt(hex.slice(1, 3), 16);
      let g = parseInt(hex.slice(3, 5), 16);
      let b = parseInt(hex.slice(5, 7), 16);
      r = Math.max(0, Math.min(255, r + amount));
      g = Math.max(0, Math.min(255, g + amount));
      b = Math.max(0, Math.min(255, b + amount));
      return '#' + [r, g, b].map(v => v.toString(16).padStart(2, '0')).join('');
    }

    function draw() {
      ctx.clearRect(0, 0, canvas.width, canvas.height);

      // Background
      ctx.fillStyle = '#0d1117';
      ctx.fillRect(0, 0, canvas.width, canvas.height);

      drawGrid();
      updateParticles();
      drawParticles();
      drawFood();
      drawSpecialFood();
      drawSnake();

      if (paused && !gameOver) {
        ctx.fillStyle = 'rgba(0,0,0,0.5)';
        ctx.fillRect(0, 0, canvas.width, canvas.height);
        ctx.fillStyle = '#fff';
        ctx.font = 'bold 28px Segoe UI';
        ctx.textAlign = 'center';
        ctx.fillText('PAUSED', canvas.width / 2, canvas.height / 2);
        ctx.font = '14px Segoe UI';
        ctx.fillStyle = '#888';
        ctx.fillText('Tap ⏸ to resume', canvas.width / 2, canvas.height / 2 + 30);
      }
    }

    function step() {
      if (paused || gameOver) return;

      dir = { ...nextDir };

      const head = { x: snake[0].x + dir.x, y: snake[0].y + dir.y };

      // Wall collision
      if (head.x < 0 || head.x >= COLS || head.y < 0 || head.y >= ROWS) {
        endGame();
        return;
      }

      // Self collision
      if (snake.some(s => s.x === head.x && s.y === head.y)) {
        endGame();
        return;
      }

      snake.unshift(head);

      let ate = false;

      // Eat food
      if (head.x === food.x && head.y === food.y) {
        ate = true;
        score += 10;
        spawnParticles(food.x, food.y, '#ff4444');
        placeFood();
        updateUI();
        checkLevelUp();
      }

      // Eat special food
      if (specialFoodActive && head.x === specialFood.x && head.y === specialFood.y) {
        ate = true;
        score += 30;
        spawnParticles(specialFood.x, specialFood.y, '#ffdd00', 14);
        specialFoodActive = false;
        updateUI();
        checkLevelUp();
      }

      if (!ate) {
        snake.pop();
      }

      // Special food timer
      if (specialFoodActive) {
        specialFoodTimer--;
        if (specialFoodTimer <= 0) {
          specialFoodActive = false;
        }
      }
    }

    function checkLevelUp() {
      const newLevel = Math.min(Math.floor(score / LEVEL_UP_SCORE) + 1, LEVELS.length);
      if (newLevel > level) {
        level = newLevel;
        restartLoop();
      }
    }

    function restartLoop() {
      clearInterval(gameLoop);
      gameLoop = setInterval(() => {
        step();
        draw();
      }, getLevelData().interval);
    }

    function endGame() {
      gameOver = true;
      clearInterval(gameLoop);

      // Explode snake
      for (const seg of snake) {
        spawnParticles(seg.x, seg.y, getLevelData().color, 4);
      }
      draw();

      setTimeout(() => {
        const overlay = document.getElementById('overlay');
        const finalScore = document.getElementById('final-score');
        const startBtn = document.getElementById('start-btn');
        overlay.classList.remove('hidden');
        document.getElementById('overlay-title').textContent = 'GAME OVER';
        document.getElementById('overlay-subtitle').textContent = score === best && score > 0
          ? '🏆 New High Score!'
          : 'Better luck next time!';
        finalScore.style.display = 'block';
        finalScore.innerHTML = 'Score: <span>' + score + '</span> | Best: <span>' + best + '</span>';
        startBtn.textContent = 'PLAY AGAIN';
      }, 600);
    }

    function startGame() {
      document.getElementById('overlay').classList.add('hidden');
      document.getElementById('final-score').style.display = 'none';
      document.getElementById('overlay-title').textContent = 'SNAKE';
      document.getElementById('overlay-subtitle').textContent = 'Swipe or use the d-pad to move.';
      document.getElementById('start-btn').textContent = 'PLAY';
      clearInterval(gameLoop);
      initGame();
      restartLoop();
    }

    // Controls
    document.getElementById('start-btn').addEventListener('click', startGame);

    document.getElementById('btn-up').addEventListener('click', () => setDir(0, -1));
    document.getElementById('btn-down').addEventListener('click', () => setDir(0, 1));
    document.getElementById('btn-left').addEventListener('click', () => setDir(-1, 0));
    document.getElementById('btn-right').addEventListener('click', () => setDir(1, 0));
    document.getElementById('btn-center').addEventListener('click', togglePause);

    function setDir(x, y) {
      if (gameOver) return;
      if (x === -dir.x && y === -dir.y) return; // no 180
      nextDir = { x, y };
      if (paused) togglePause();
    }

    function togglePause() {
      if (gameOver) return;
      paused = !paused;
      if (!paused) draw();
    }

    // Keyboard
    document.addEventListener('keydown', e => {
      switch (e.key) {
        case 'ArrowUp':    case 'w': setDir(0, -1); break;
        case 'ArrowDown':  case 's': setDir(0, 1);  break;
        case 'ArrowLeft':  case 'a': setDir(-1, 0); break;
        case 'ArrowRight': case 'd': setDir(1, 0);  break;
        case 'p': case 'Escape': togglePause(); break;
        case 'Enter': if (gameOver || document.getElementById('overlay').style.opacity !== '0') startGame(); break;
      }
    });

    // Touch swipe
    let touchStartX = 0, touchStartY = 0;

    canvas.addEventListener('touchstart', e => {
      e.preventDefault();
      touchStartX = e.touches[0].clientX;
      touchStartY = e.touches[0].clientY;
    }, { passive: false });

    canvas.addEventListener('touchend', e => {
      e.preventDefault();
      const dx = e.changedTouches[0].clientX - touchStartX;
      const dy = e.changedTouches[0].clientY - touchStartY;
      const minSwipe = 20;

      if (Math.abs(dx) < minSwipe && Math.abs(dy) < minSwipe) return;

      if (Math.abs(dx) > Math.abs(dy)) {
        setDir(dx > 0 ? 1 : -1, 0);
      } else {
        setDir(0, dy > 0 ? 1 : -1);
      }
    }, { passive: false });

    // Initial draw
    initGame();
    draw();
  </script>
